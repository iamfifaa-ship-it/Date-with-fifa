<script>

const messages = [
"ใจร้ายจัง 🥺",
"ไม่ไปจริงเหรอ 😭",
"เสียใจมาก 💔",
"อยากร้องไห้แล้วนะ 🥹",
"งั้นขอลองถามใหม่อีกที...",
"ตอบผิดหรือเปล่า 🤔",
"ระบบตรวจพบความใจร้าย 🚨",
"แน่ใจนะ เดี๋ยวเสียโอกาสกินของอร่อยนะ 🍜"
];

let noCount = 0;
let yesScale = 1;

const yesBtn = document.getElementById("yesBtn");
const noBtn = document.getElementById("noBtn");

noBtn.addEventListener("click",()=>{

    noCount++;

    document.getElementById("message").innerText =
        messages[Math.floor(Math.random()*messages.length)];

    yesScale += 0.15;
    yesBtn.style.transform = `scale(${yesScale})`;

    if(noCount === 10){
        alert("🚨 การปฏิเสธเกินกำหนด กรุณาพิจารณาคำตอบใหม่อีกครั้ง");
    }

    if(noCount >= 15){

        const x = Math.random() * (window.innerWidth - 150);
        const y = Math.random() * (window.innerHeight - 100);

        noBtn.style.position = "fixed";
        noBtn.style.left = x + "px";
        noBtn.style.top = y + "px";
    }

    if(noCount >= 30){

        document.body.innerHTML = `
        <div style="
        display:flex;
        justify-content:center;
        align-items:center;
        flex-direction:column;
        width:100vw;
        height:100vh;
        background:black;
        color:white;
        text-align:center;
        font-family:sans-serif;">
        <h1>🥲 ยอมแล้วครับ</h1>
        <p>เป็นเพื่อนกันก็ได้</p>
        <button onclick="location.reload()"
        style="padding:15px 25px;margin-top:20px;">
        ล้อเล่น กดไปเดทเถอะ
        </button>
        </div>
        `;
    }
});

yesBtn.addEventListener("click",()=>{
    document.getElementById("page1").classList.add("hidden");
    document.getElementById("page2").classList.remove("hidden");
});

function submitDate(){

    const date = document.getElementById("date").value;
    const time = document.getElementById("time").value;
    const activity = document.getElementById("activity").value;

    document.getElementById("page2").classList.add("hidden");
    document.getElementById("page3").classList.remove("hidden");

    document.getElementById("summary").innerHTML = `
    <h2>📅 ${date}</h2>
    <h2>🕒 ${time}</h2>
    <h2>🎯 ${activity}</h2>
    <br>
    <p>ขอบคุณที่ไม่กดปฏิเสธจนถึงครั้งที่ ${noCount} 😂</p>
    `;

    createHearts();
}

function createHearts(){

    setInterval(()=>{

        const heart = document.createElement("div");

        heart.className = "heart";
        heart.innerHTML = "❤️";

        heart.style.left =
            Math.random()*window.innerWidth+"px";

        document.body.appendChild(heart);

        setTimeout(()=>{
            heart.remove();
        },5000);

    },250);
}

document.getElementById("secretBtn").addEventListener("click",()=>{

    alert(
`ข้อดีของการไปเดทกับผม

🍜 เลี้ยงข้าวได้

📸 ถ่ายรูปให้ได้
👨‍🍳 ทำอาหารให้กินได้
🥹 มุกอาจจะฝืด แต่ตั้งใจนะ`
    );

});
